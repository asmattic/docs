*args and **kwargs¶

Parsing a JSON File¶

# File is an array of onjects that are only one level deep

import json

all_things_text = []

with open('things.txt') as json_things:
        things = json.load(json_things)
        for thing in things:
                all_things_text.append(thing['text'])
                print('\n Thing %s \n %s \n' % (thing['id'], thing['text']))
